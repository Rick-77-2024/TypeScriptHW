// У нас 1000 рублей и нам нужно эти 1000 рублей перевести в доллары
// Курс доллара допустим 85 рублей за 1$
// соответственно нам нужно rub поделить на wellUSD

 function currencyExchange(amount, rate) {
     if (typeof amount !== 'number' || typeof rate !== 'number') {
         throw new Error('Amount and rate must be numbers');
     }
     if (amount < 0 || rate <= 0) {
         throw new Error('Amount must be non-negative and rate must be positive');
     }
     return amount * rate;
 }

 // Приклад використання:
 const usdToEurRate = 0.92;
 const dollars = 100;
 const euros = currencyExchange(dollars, usdToEurRate);
 console.log(`${dollars} USD = ${euros.toFixed(2)} EUR`);
